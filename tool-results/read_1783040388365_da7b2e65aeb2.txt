     3→/**
     4→ * Faya Admin Portal — Merchant Detail View (Full Page Profile)
     5→ *
     6→ * A proper profile page (NOT a slide-out sheet) showing EVERYTHING connected
     7→ * to a merchant under one roof. Replaces the previous Sheet pattern with a
     8→ * full-width, scrollable page that has a back button to return to the list.
     9→ *
    10→ * Layout:
    11→ *   A. Back button bar at the top → setView("merchants")
    12→ *   B. Profile header card (full-width):
    13→ *        - Left: business avatar + trading name + legal name + merchant code + country
    14→ *        - Center: KYB / account status / risk badges + platform chips
    15→ *        - Right: action buttons (Approve KYB / Restrict / Suspend / Reactivate)
    16→ *                 with AlertDialog confirmations
    17→ *        - Bottom row: 6 quick-stat cards (POS Staff, Terminals, Phone POS,
    18→ *                       Monthly Volume, Open Disputes, Open Tickets)
    19→ *   C. Tabs section (full width):
    20→ *        1. Overview         7. Disputes
    21→ *        2. POS Staff        8. Documents
    22→ *        3. Terminals        9. KYB Cases
    23→ *        4. POS Requests    10. Support
    24→ *        5. Transactions    11. Risk & Alerts
    25→ *        6. Settlements     12. Activity & Audit
    26→ *
    27→ * POS Device Request approval rule (per spec):
    28→ *   - canBeApproved = NFC || cardReader || swipe
    29→ *   - If false → Approve disabled + Auto-decline action sets status="auto_declined"
    30→ *   - Amber warnings for failed device integrity or no screen lock
    31→ *
    32→ * Subscriptions: a single useEffect at the page level subscribes to all 11
    33→ * collections (PosStaff, Terminals, PosDeviceRequests, Transactions,
    34→ * Settlements, Disputes, Documents, KybCases, Tickets, FraudAlerts, AuditLogs)
    35→ * and passes filtered slices down to each tab. This guarantees the count
    36→ * badges in tab labels are accurate even before a tab is opened.
    37→ *
    38→ * Audit action keys (via logAudit):
    39→ *   merchant.approve_kyb / merchant.reject_kyb / merchant.restrict /
    40→ *   merchant.suspend / merchant.reactivate /
    41→ *   pos_staff.suspend / pos_staff.reactivate / pos_staff.reset_pin /
    42→ *   pos_staff.force_logout /
    43→ *   terminal.activate / terminal.block /
    44→ *   pos_device.approve / pos_device.decline / pos_device.auto_decline /
    45→ *   settlement.retry /
    46→ *   dispute.request_evidence / dispute.update_status /

   440→    <Badge variant="secondary" className={cn("text-[10px]", cls)}>
   441→      {s.label}
   442→    </Badge>
   443→  );

   446→/** Match a name against the merchant's legal/trading names. */
   447→function nameMatchesMerchant(
   448→  name: string | null | undefined,
   449→  merchant: Merchant,
   450→): boolean {
   451→  if (!name) return false;
   452→  return name === merchant.legalName || name === merchant.tradingName;
   453→}

   621→  function applyMerchantAction(action: MerchantAction) {
   622→    if (!staff || !merchant) return;

   624→    let patch: Partial<Merchant> = {};

   627→    if (action === "restrict") {
   628→      patch = { status: "restricted", updatedAt: Date.now() };
   629→      after = `${merchant.kybStatus} / restricted`;
   630→    } else if (action === "suspend") {
   631→      patch = { status: "suspended", updatedAt: Date.now() };
   632→      after = `${merchant.kybStatus} / suspended`;
   633→    } else if (action === "reactivate") {
   634→      patch = { status: "active", updatedAt: Date.now() };
   635→      after = `${merchant.kybStatus} / active`;
   636→    } else if (action === "approve_kyb") {
   637→      patch = { kybStatus: "approved", status: "active", updatedAt: Date.now() };
   638→      after = "approved / active";
   639→    } else if (action === "reject_kyb") {
   640→      patch = { kybStatus: "rejected", status: "restricted", updatedAt: Date.now() };
   641→      after = "rejected / restricted";
   642→    }
   643→
   644→    adminData.updateMerchant(merchant.id, patch);
   645→    logAudit(
   646→      {
   647→        staffId: staff.id,
   648→        staffName: `${staff.firstName} ${staff.lastName}`,
   649→        department: staff.departmentId,
   650→        role: staff.roleId,
   651→      },
   652→      `merchant.${action}`,
   653→      "merchant",
   654→      merchant.id,
   655→      {
   656→        countryCode: merchant.countryCode,
   657→        beforeValue: before,
   658→        afterValue: after,
   659→      },
   660→    );
   661→    toast.success(`${action.replace(/_/g, " ")} — ${merchant.tradingName}`, {
   662→      description: `${merchant.merchantCode} · ${before} → ${after}`,
   663→    });
   664→    setConfirmAction(null);
   665→  }
