   147→   147→  /* ----------------- Linked entity lookups (KYC → Consumer, KYB → Merchant) ----------------- */
   148→   148→  // Match by kycCaseId first; when multiple consumers share a case id (e.g. spouses
   149→   149→  // on a joint application), disambiguate by exact customerName match. Fall back to
   150→   150→  // name-only match if kycCaseId is null on both sides.
   151→   151→  function findConsumerForKyc(k: KycCase): Consumer | null {
   152→   152→    if (!consumers.length) return null;
   153→   153→    const fullName = `${k.customerName}`.trim().toLowerCase();
   154→   154→    const byCaseId = consumers.filter((c) => c.kycCaseId === k.id);
   155→   155→    if (byCaseId.length === 1) return byCaseId[0];
   156→   156→    if (byCaseId.length > 1) {
   157→   157→      const byName = byCaseId.find(
   158→   158→        (c) => `${c.firstName} ${c.lastName}`.trim().toLowerCase() === fullName,
   159→   159→      );
   160→   160→      return byName ?? byCaseId[0];
   161→   161→    }
   162→   162→    // No case-id link — try exact name match as a fallback (older case with cleared link)
   163→   163→    return (
   164→   164→      consumers.find(
   165→   165→        (c) => `${c.firstName} ${c.lastName}`.trim().toLowerCase() === fullName,
   166→   166→      ) ?? null
   167→   167→    );
   168→   168→  }
   169→   169→
   170→   170→  function findMerchantForKyb(k: KybCase): Merchant | null {
   171→   171→    if (!merchants.length) return null;
   172→   172→    const name = `${k.merchantName}`.trim().toLowerCase();
   173→   173→    const byCaseId = merchants.filter((m) => m.kybCaseId === k.id);
   174→   174→    if (byCaseId.length === 1) return byCaseId[0];
   175→   175→    if (byCaseId.length > 1) {
   176→   176→      const byName =
   177→   177→        byCaseId.find((m) => m.legalName.trim().toLowerCase() === name) ??
   178→   178→        byCaseId.find((m) => m.tradingName.trim().toLowerCase() === name);
   179→   179→      return byName ?? byCaseId[0];
   180→   180→    }
   181→   181→    // No case-id link — try name match against legal / trading name
   182→   182→    return (
   183→   183→      merchants.find((m) => m.legalName.trim().toLowerCase() === name) ??
   184→   184→      merchants.find((m) => m.tradingName.trim().toLowerCase() === name) ??
   185→   185→      null
   186→   186→    );
   187→   187→  }

   268→   268→  /* --------------------------- Lookups --------------------------- */

   277→   277→
   278→   278→  /* ----------------------- KYC mutations ------------------------- */
   279→   279→  function approveKyc(k: KycCase) {
   280→   280→    if (!actor) return;

   288→   288→    // Live-sync the linked consumer record so the Consumer App sees the new status

   290→   290→    if (consumer) {
   291→   291→      adminData.updateConsumer(consumer.id, {
   292→   292→        kycStatus: "approved",
   293→   293→        status: "active",
   294→   294→        updatedAt: Date.now(),

   307→   307→        description: `Case ${k.id} · ${countryName(k.countryCode)} · no linked consumer record.`,

   309→   309→    }

   326→   326→        status: "restricted",

   335→   335→      toast.error(`KYC rejected: ${k.customerName}`, {
   336→   336→        description: `Case ${k.id} · Consumer ${consumer.consumerCode} restricted.`,

   338→   338→    } else {
   339→   339→      toast.error(`KYC rejected: ${k.customerName}`, { description: `Case ${k.id}` });

   343→   343→  function escalateKyc(k: KycCase) {

   347→   347→      countryCode: k.countryCode,
   348→   348→      beforeValue: k.status,
   349→   349→      afterValue: "escalated",
   350→   350→    });
   351→   351→    toast.warning(`KYC escalated: ${k.customerName}`, {
   352→   352→      description: "Moved to Sanctions/PEP review queue",

   354→   354→    setSelectedKyc(null);
   355→   355→  }
   356→   356→  function requestDocsKyc(k: KycCase) {
   357→   357→    if (!actor) return;
   358→   358→    logAudit(actor, "kyc.request_documents", "kyc_case", k.id, {
   359→   359→      countryCode: k.countryCode,
   360→   360→    });
   361→   361→    toast.info(`Document request sent: ${k.customerName}`, {
   362→   362→      description: `Notifying customer to upload: ${k.requiredDocuments
   363→   363→        .map((d) => d.replace(/_/g, " "))
   364→   364→        .join(", ")}`,
   365→   365→    });
   366→   366→  }
   367→   367→  function assignReviewerKyc(k: KycCase) {
   368→   368→    if (!actor || !currentStaff) return;
   369→   369→    const newStatus: KycCase["status"] = k.status === "pending" ? "in_review" : k.status;
   370→   370→    adminData.updateKyc(k.id, { assignedReviewer: currentStaff.id, status: newStatus });
   371→   371→    logAudit(actor, "kyc.assign", "kyc_case", k.id, {
   372→   372→      countryCode: k.countryCode,
   373→   373→      afterValue: `${currentStaff.firstName} ${currentStaff.lastName}`,
   374→   374→    });
   375→   375→    toast.success(`Assigned to you: ${k.customerName}`, {
   376→   376→      description: `You are now the reviewer for case ${k.id}`,
   377→   377→    });
   378→   378→    setSelectedKyc({ ...k, assignedReviewer: currentStaff.id, status: newStatus });
   379→   379→  }
   380→   380→
   381→   381→  /* ----------------------- KYB mutations ------------------------- */
   382→   382→  function approveKyb(k: KybCase) {
   383→   383→    if (!actor) return;
   384→   384→    const reviewer = k.assignedReviewer ?? currentStaff?.id ?? null;
   385→   385→    adminData.updateKyb(k.id, { status: "approved", assignedReviewer: reviewer });
   386→   386→    logAudit(actor, "kyb.approve", "kyb_case", k.id, {
   387→   387→      countryCode: k.countryCode,
   388→   388→      beforeValue: k.status,
   389→   389→      afterValue: "approved",
   390→   390→    });
   391→   391→    // Live-sync the linked merchant record so the Merchant App sees the new status
   392→   392→    const merchant = findMerchantForKyb(k);
   393→   393→    if (merchant) {
   394→   394→      adminData.updateMerchant(merchant.id, {
   395→   395→        kybStatus: "approved",
   396→   396→        status: "active",
   397→   397→        updatedAt: Date.now(),

   401→   401→        beforeValue: `${merchant.kybStatus} / ${merchant.status}`,
   402→   402→        afterValue: "approved / active",
   403→   403→        reason: `KYB case ${k.id} approved`,
   404→   404→      });
   405→   405→      toast.success(`KYB approved: ${k.merchantName}`, {
   406→   406→        description: `Case ${k.id} · Merchant ${merchant.merchantCode} activated in real-time.`,
   407→   407→      });
   408→   408→    } else {
   409→   409→      toast.success(`KYB approved: ${k.merchantName}`, {
   410→   410→        description: `Case ${k.id} · ${countryName(k.countryCode)} · no linked merchant record.`,
   411→   411→      });
   412→   412→    }
   413→   413→    setSelectedKyb(null);

   415→   415→  function rejectKyb(k: KybCase) {
   416→   416→    if (!actor) return;
   417→   417→    const reviewer = k.assignedReviewer ?? currentStaff?.id ?? null;
   418→   418→    adminData.updateKyb(k.id, { status: "rejected", assignedReviewer: reviewer });
   419→   419→    logAudit(actor, "kyb.reject", "kyb_case", k.id, {
   420→   420→      countryCode: k.countryCode,
   421→   421→      beforeValue: k.status,
   422→   422→      afterValue: "rejected",
   423→   423→    });
   424→   424→    // Live-sync the linked merchant record — rejected cases restrict the account
   425→   425→    const merchant = findMerchantForKyb(k);
   426→   426→    if (merchant) {
   427→   427→      adminData.updateMerchant(merchant.id, {
   428→   428→        kybStatus: "rejected",
   429→   429→        status: "restricted",
   430→   430→        updatedAt: Date.now(),
   431→   431→      });
   432→   432→      logAudit(actor, "merchant.kyb_rejected", "merchant", merchant.id, {
   433→   433→        countryCode: merchant.countryCode,
   434→   434→        beforeValue: `${merchant.kybStatus} / ${merchant.status}`,
   435→   435→        afterValue: "rejected / restricted",
   436→   436→        reason: `KYB case ${k.id} rejected`,
   437→   437→      });
   438→   438→      toast.error(`KYB rejected: ${k.merchantName}`, {
   439→   439→        description: `Case ${k.id} · Merchant ${merchant.merchantCode} restricted.`,
   440→   440→      });
   441→   441→    } else {
   442→   442→      toast.error(`KYB rejected: ${k.merchantName}`, { description: `Case ${k.id}` });
   443→   443→    }
