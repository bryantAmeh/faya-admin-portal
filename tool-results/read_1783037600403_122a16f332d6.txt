   130‚Üí   130‚Üí/**
   131‚Üí   131‚Üí * Tracks whether we've fallen back to local mode.
   132‚Üí   132‚Üí * Once true, all subsequent operations use the local store.

   134‚Üí   134‚Üílet useLocalMode = false;
   135‚Üí   135‚Üílet firestoreCheckPromise: Promise<boolean> | null = null;
   136‚Üí   136‚Üí
   137‚Üí   137‚Üí/**
   138‚Üí   138‚Üí * Check whether Firestore is accessible. Returns true if Firestore works,
   139‚Üí   139‚Üí * false if we should use local mode. Result is cached.
   140‚Üí   140‚Üí */
   141‚Üí   141‚Üíasync function checkFirestoreAccess(): Promise<boolean> {
   142‚Üí   142‚Üí  if (useLocalMode) return false;
   143‚Üí   143‚Üí  if (firestoreCheckPromise) return firestoreCheckPromise;
   144‚Üí   144‚Üí  firestoreCheckPromise = (async () => {
   145‚Üí   145‚Üí    const d = db();
   146‚Üí   146‚Üí    if (!d) {
   147‚Üí   147‚Üí      useLocalMode = true;
   148‚Üí   148‚Üí      localStore.active = true;
   149‚Üí   149‚Üí      return false;
   150‚Üí   150‚Üí    }
   151‚Üí   151‚Üí    try {
   152‚Üí   152‚Üí      // Try a tiny read ‚Äî if it fails with permission-denied, fall back.
   153‚Üí   153‚Üí      await getDocs(collection(d, COLLECTIONS.meta));
   154‚Üí   154‚Üí      return true;
   155‚Üí   155‚Üí    } catch (e) {
   156‚Üí   156‚Üí      const code = (e as { code?: string })?.code ?? "";
   157‚Üí   157‚Üí      if (
   158‚Üí   158‚Üí        code === "permission-denied" ||
   159‚Üí   159‚Üí        code === "unavailable" ||
   160‚Üí   160‚Üí        code === "unimplemented" ||
   161‚Üí   161‚Üí        code.includes("permission")
   162‚Üí   162‚Üí      ) {
   163‚Üí   163‚Üí        console.warn(
   164‚Üí   164‚Üí          "[admin-data] Firestore access denied ‚Äî switching to local mode. Configure Firestore security rules to enable cloud sync.",
   165‚Üí   165‚Üí          code,
   166‚Üí   166‚Üí        );
   167‚Üí   167‚Üí        useLocalMode = true;
   168‚Üí   168‚Üí        localStore.active = true;
   169‚Üí   169‚Üí        return false;
   170‚Üí   170‚Üí      }
   171‚Üí   171‚Üí      // Other errors ‚Äî also fall back to keep the portal usable
   172‚Üí   172‚Üí      console.warn("[admin-data] Firestore check failed, using local mode:", code || e);
   173‚Üí   173‚Üí      useLocalMode = true;
   174‚Üí   174‚Üí      localStore.active = true;
   175‚Üí   175‚Üí      return false;
   176‚Üí   176‚Üí    }
   177‚Üí   177‚Üí  })();
   178‚Üí   178‚Üí  return firestoreCheckPromise;
   179‚Üí   179‚Üí}

   183‚Üí   183‚Üí/**

   196‚Üí   196‚Üí    try {

   260‚Üí   260‚Üí/**
   261‚Üí   261‚Üí * Subscribe to a collection with optional query constraints.
   262‚Üí   262‚Üí * Falls back to local store on Firestore access failure.
   263‚Üí   263‚Üí */
   264‚Üí   264‚Üíexport function subscribe<T extends { id: string }>(
   265‚Üí   265‚Üí  colName: string,
   266‚Üí   266‚Üí  cb: (items: T[]) => void,
   267‚Üí   267‚Üí  ...constraints: QueryConstraint[]
   268‚Üí   268‚Üí): Unsubscribe {
   269‚Üí   269‚Üí  const localCol = LOCAL_COLLECTION_MAP[colName];
   270‚Üí   270‚Üí  if (!localCol) {
   271‚Üí   271‚Üí    console.error(`[subscribe] unknown collection: ${colName}`);
   272‚Üí   272‚Üí    return () => {};
   273‚Üí   273‚Üí  }
   274‚Üí   274‚Üí
   275‚Üí   275‚Üí  // If we already know we're in local mode, subscribe locally immediately
   276‚Üí   276‚Üí  if (useLocalMode) {
   277‚Üí   277‚Üí    return localStore.subscribe<T>(localCol, cb);
   278‚Üí   278‚Üí  }
   279‚Üí   279‚Üí
   280‚Üí   280‚Üí  // Otherwise try Firestore; on error, switch to local
   281‚Üí   281‚Üí  let unsub: Unsubscribe | null = null;
   282‚Üí   282‚Üí  let switchedToLocal = false;
   283‚Üí   283‚Üí
   284‚Üí   284‚Üí  const d = db();
   285‚Üí   285‚Üí  if (!d) {
   286‚Üí   286‚Üí    useLocalMode = true;
   287‚Üí   287‚Üí    localStore.active = true;
   288‚Üí   288‚Üí    return localStore.subscribe<T>(localCol, cb);
   289‚Üí   289‚Üí  }
   290‚Üí   290‚Üí
   291‚Üí   291‚Üí  try {
   292‚Üí   292‚Üí    const q =
   293‚Üí   293‚Üí      constraints.length
   294‚Üí   294‚Üí        ? query(collection(d, colName), ...constraints)
   295‚Üí   295‚Üí        : collection(d, colName);
   296‚Üí   296‚Üí    unsub = onSnapshot(
   297‚Üí   297‚Üí      q,
   298‚Üí   298‚Üí      (snap) => {
   299‚Üí   299‚Üí        const items = snap.docs.map((d) => {
   300‚Üí   300‚Üí          const data = d.data() as DocumentData;
   301‚Üí   301‚Üí          const { _serverTime, ...rest } = data as Record<string, unknown>;
   302‚Üí   302‚Üí          return { ...(rest as T), id: (rest.id as string) ?? d.id };
   303‚Üí   303‚Üí        });
   304‚Üí   304‚Üí        cb(items);
   305‚Üí   305‚Üí      },
   306‚Üí   306‚Üí      (err) => {
   307‚Üí   307‚Üí        const code = (err as { code?: string })?.code ?? "";
   308‚Üí   308‚Üí        if (
   309‚Üí   309‚Üí          code === "permission-denied" ||
   310‚Üí   310‚Üí          code === "unavailable" ||
   311‚Üí   311‚Üí          code.includes("permission") ||
   312‚Üí   312‚Üí          !useLocalMode
   313‚Üí   313‚Üí        ) {
   314‚Üí   314‚Üí          if (!useLocalMode) {
   315‚Üí   315‚Üí            console.warn(
   316‚Üí   316‚Üí              `[subscribe:${colName}] Firestore error (${code || "unknown"}), switching to local mode`,
   317‚Üí   317‚Üí            );
   318‚Üí   318‚Üí            useLocalMode = true;
   319‚Üí   319‚Üí            localStore.active = true;
   320‚Üí   320‚Üí          }
   321‚Üí   321‚Üí          if (!switchedToLocal) {
   322‚Üí   322‚Üí            switchedToLocal = true;
   323‚Üí   323‚Üí            localStore.subscribe<T>(localCol, cb);
   324‚Üí   324‚Üí          }
   325‚Üí   325‚Üí        }
   326‚Üí   326‚Üí      },
   327‚Üí   327‚Üí    );
   328‚Üí   328‚Üí    return () => {
   329‚Üí   329‚Üí      unsub?.();

   340‚Üí   340‚Üí  const localCol = LOCAL_COLLECTION_MAP[colName];

   342‚Üí   342‚Üí

   346‚Üí   346‚Üí      unsub = localStore.subscribe<T>(localCol, (items) => {

   350‚Üí   350‚Üí      });

   356‚Üí   356‚Üí  try {

   359‚Üí   359‚Üí      const data = d.data() as DocumentData;

   366‚Üí   366‚Üí      useLocalMode = true;

   368‚Üí   368‚Üí      return new Promise((resolve) => {
   369‚Üí   369‚Üí        let unsub: (() => void) | null = null;
   370‚Üí   370‚Üí        unsub = localStore.subscribe<T>(localCol, (items) => {

   372‚Üí   372‚Üí          resolve(items);
   373‚Üí   373‚Üí        });

   375‚Üí   375‚Üí    }
