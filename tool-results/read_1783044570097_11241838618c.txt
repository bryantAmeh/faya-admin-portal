   272→   272→
   273→   273→export function UserDetailView({ consumers, countries }: UserDetailViewProps) {
   274→   274→  const { staff } = useAuth();
   275→   275→  const { selectedUserId, setView } = usePortalStore();
   276→   276→  const [confirmAction, setConfirmAction] = useState<{
   277→   277→    consumer: Consumer;
   278→   278→    action: "restrict" | "suspend" | "reactivate";
   279→   279→  } | null>(null);
   280→   280→
   281→   281→  const consumer = useMemo(
   282→   282→    () => consumers.find((c) => c.id === selectedUserId) ?? null,
   283→   283→    [consumers, selectedUserId],
   284→   284→  );
   285→   285→
   286→   286→  function countryName(code: string): string {
   287→   287→    return countries.find((c) => c.countryCode === code)?.countryName ?? code;
   288→   288→  }
   289→   289→
   290→   290→  async function applyConsumerAction(
   291→   291→    c: Consumer,
   292→   292→    action: "restrict" | "suspend" | "reactivate",
   293→   293→  ) {
   294→   294→    if (!staff) return;
   295→   295→    const newStatus: ConsumerStatus =
   296→   296→      action === "restrict"
   297→   297→        ? "restricted"
   298→   298→        : action === "suspend"
   299→   299→          ? "suspended"
   300→   300→          : "active";
   301→   301→    const before = c.status;
   302→   302→    await adminData.updateConsumer(c.id, {
   303→   303→      status: newStatus,
   304→   304→      updatedAt: Date.now(),
   305→   305→    });
   306→   306→    logAudit(
   307→   307→      {
   308→   308→        staffId: staff.id,
   309→   309→        staffName: `${staff.firstName} ${staff.lastName}`,
   310→   310→        department: staff.departmentId,
   311→   311→        role: staff.roleId,
   312→   312→      },
   313→   313→      `consumer.${action}`,
   314→   314→      "consumer",
   315→   315→      c.id,
   316→   316→      {
   317→   317→        countryCode: c.countryCode,
   318→   318→        beforeValue: before,
   319→   319→        afterValue: newStatus,
   320→   320→        reason: `${action} consumer ${c.consumerCode}`,
   321→   321→      },
   322→   322→    );
   323→   323→    toast.success(
   324→   324→      `${c.firstName} ${c.lastName} ${action === "reactivate" ? "reactivated" : action + "ed"} successfully`,
   325→   325→    );
   326→   326→  }
   327→   327→
   328→   328→  // Not-found fallback
   329→   329→  if (!consumer) {
   330→   330→    return (
   331→   331→      <ViewContainer>
   332→   332→        <SonnerToaster richColors position="top-right" />
   333→   333→        <Button variant="outline" size="sm" onClick={() => setView("users")}>
   334→   334→          <ArrowLeft className="size-4 mr-1" /> Back to Users
   335→   335→        </Button>
   336→   336→        <Card>
   337→   337→          <CardContent className="p-0">
   338→   338→            <EmptyState
   339→   339→              icon={UsersIcon}
   340→   340→              title="Consumer not found"
   341→   341→              description="The selected consumer no longer exists. Pick another from the users list."
   342→   342→            />
   343→   343→          </CardContent>
   344→   344→        </Card>
   345→   345→      </ViewContainer>
   346→   346→    );
   347→   347→  }
   348→   348→
   349→   349→  return (
   350→   350→    <ViewContainer className="space-y-4">
   351→   351→      <SonnerToaster richColors position="top-right" />
   352→   352→
   353→   353→      {/* A. Back button bar */}
   354→   354→      <div className="flex items-center justify-between">
   355→   355→        <Button variant="outline" size="sm" onClick={() => setView("users")}>
   356→   356→          <ArrowLeft className="size-4 mr-1" /> Back to Users
   357→   357→        </Button>
   358→   358→        <Badge variant="outline" className="text-[11px] text-emerald-700 border-emerald-300 dark:text-emerald-400 dark:border-emerald-700">
   359→   359→          Full Profile View
   360→   360→        </Badge>
   361→   361→      </div>
   362→   362→
   363→   363→      {/* B. Profile header — keyed by id so all subscriptions/tabs reset on switch */}
   364→   364→      <ProfileHeader
   365→   365→        key={consumer.id}
   366→   366→        consumer={consumer}
   367→   367→        countries={countries}
   368→   368→        countryName={countryName}
   369→   369→        staff={staff}
   370→   370→        onAction={(action) => setConfirmAction({ consumer, action })}
   371→   371→      />
   372→   372→
   373→   373→      {/* C. Tabs */}
   374→   374→      <ProfileTabs
   375→   375→        key={`tabs-${consumer.id}`}
   376→   376→        consumer={consumer}
   377→   377→        countries={countries}
   378→   378→        countryName={countryName}
   379→   379→        staff={staff}
   380→   380→      />
   381→   381→
   382→   382→      {/* Confirm action dialog */}
   383→   383→      <AlertDialog
   384→   384→        open={!!confirmAction}
   385→   385→        onOpenChange={(o) => !o && setConfirmAction(null)}
   386→   386→      >
   387→   387→        <AlertDialogContent>
   388→   388→          <AlertDialogHeader>
   389→   389→            <AlertDialogTitle>
   390→   390→              {confirmAction?.action === "restrict" && "Restrict consumer"}
   391→   391→              {confirmAction?.action === "suspend" && "Suspend consumer"}
   392→   392→              {confirmAction?.action === "reactivate" && "Reactivate consumer"}
   393→   393→            </AlertDialogTitle>
